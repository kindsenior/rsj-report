%% alias
\newcommand{\doublealias}[5]{
  \ifcsname#1\endcsname
    \ifcsname#2\endcsname
      \message{[doublealias] Do nothing because #1 and #2 is already defined!!!}
      \else
      \newcommand{#4}[1]{{#3{##1}}}
    \fi
  \else
    \ifcsname#2\endcsname
      \newcommand{#3}[1]{{#4{##1}}}
    \else
      \newcommand{#3}[1]{#5}
      \newcommand{#4}[1]{#5}
    \fi
  \fi
}
